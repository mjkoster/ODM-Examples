  namespace {
    ex https://onedm.org/exploratory/
    pg https://onedm.org/playground/
  } 
  defaultNamespace ex

  sdfObject {
    ovenMode {
      label "Oven Mode"
      required [
        sdfProperty/ovenMode
        sdfProperty/supportedOvenModes
        sdfAction/setOvenMode
      ]
      sdfProperty {
        ovenMode {
          Ref ex:#/sdfData/OvenMode
        }
        supportedOvenModes {
          type array
          items {
            Ref ex:#/sdfData/OvenMode
          }
        }
      } 
      sdfAction {
        setOvenMode { 
          description "Action to set the oven mode"
          label "Set Oven Mode"
          inputData {
            required [ ovenMode ]
            fields {
              ovenMode {
                Ref ex:#/sdfData/OvenMode
              }
            }
          }
        }
      }
    }
  }
}
